\DoxyCodeLine{21   \textcolor{comment}{/* This file is used to define the FreeType module error codes.          */}}
\DoxyCodeLine{22   \textcolor{comment}{/*                                                                       */}}
\DoxyCodeLine{23   \textcolor{comment}{/* If the macro FT\_CONFIG\_OPTION\_USE\_MODULE\_ERRORS in `ftoption.h' is    */}}
\DoxyCodeLine{24   \textcolor{comment}{/* set, the lower byte of an error value identifies the error code as    */}}
\DoxyCodeLine{25   \textcolor{comment}{/* usual.  In addition, the higher byte identifies the module.  For      */}}
\DoxyCodeLine{26   \textcolor{comment}{/* example, the error `FT\_Err\_Invalid\_File\_Format' has value 0x0003, the */}}
\DoxyCodeLine{27   \textcolor{comment}{/* error `TT\_Err\_Invalid\_File\_Format' has value 0x1303, the error        */}}
\DoxyCodeLine{28   \textcolor{comment}{/* `T1\_Err\_Invalid\_File\_Format' has value 0x1403, etc.                   */}}
\DoxyCodeLine{29   \textcolor{comment}{/*                                                                       */}}
\DoxyCodeLine{30   \textcolor{comment}{/* Note that `FT\_Err\_Ok', `TT\_Err\_Ok', etc. are always equal to zero,    */}}
\DoxyCodeLine{31   \textcolor{comment}{/* including the high byte.                                              */}}
\DoxyCodeLine{32   \textcolor{comment}{/*                                                                       */}}
\DoxyCodeLine{33   \textcolor{comment}{/* If FT\_CONFIG\_OPTION\_USE\_MODULE\_ERRORS isn't set, the higher byte of   */}}
\DoxyCodeLine{34   \textcolor{comment}{/* an error value is set to zero.                                        */}}
\DoxyCodeLine{35   \textcolor{comment}{/*                                                                       */}}
\DoxyCodeLine{36   \textcolor{comment}{/* To hide the various `XXX\_Err\_' prefixes in the source code, FreeType  */}}
\DoxyCodeLine{37   \textcolor{comment}{/* provides some macros in `fttypes.h'.                                  */}}
\DoxyCodeLine{38   \textcolor{comment}{/*                                                                       */}}
\DoxyCodeLine{39   \textcolor{comment}{/*   FT\_ERR( err )                                                       */}}
\DoxyCodeLine{40   \textcolor{comment}{/*     Add current error module prefix (as defined with the              */}}
\DoxyCodeLine{41   \textcolor{comment}{/*     `FT\_ERR\_PREFIX' macro) to `err'.  For example, in the BDF module  */}}
\DoxyCodeLine{42   \textcolor{comment}{/*     the line                                                          */}}
\DoxyCodeLine{43   \textcolor{comment}{/*                                                                       */}}
\DoxyCodeLine{44   \textcolor{comment}{/*       error = FT\_ERR( Invalid\_Outline );                              */}}
\DoxyCodeLine{45   \textcolor{comment}{/*                                                                       */}}
\DoxyCodeLine{46   \textcolor{comment}{/*     expands to                                                        */}}
\DoxyCodeLine{47   \textcolor{comment}{/*                                                                       */}}
\DoxyCodeLine{48   \textcolor{comment}{/*       error = BDF\_Err\_Invalid\_Outline;                                */}}
\DoxyCodeLine{49   \textcolor{comment}{/*                                                                       */}}
\DoxyCodeLine{50   \textcolor{comment}{/*     For simplicity, you can always use `FT\_Err\_Ok' directly instead   */}}
\DoxyCodeLine{51   \textcolor{comment}{/*     of `FT\_ERR( Ok )'.                                                */}}
\DoxyCodeLine{52   \textcolor{comment}{/*                                                                       */}}
\DoxyCodeLine{53   \textcolor{comment}{/*   FT\_ERR\_EQ( errcode, err )                                           */}}
\DoxyCodeLine{54   \textcolor{comment}{/*   FT\_ERR\_NEQ( errcode, err )                                          */}}
\DoxyCodeLine{55   \textcolor{comment}{/*     Compare error code `errcode' with the error `err' for equality    */}}
\DoxyCodeLine{56   \textcolor{comment}{/*     and inequality, respectively.  Example:                           */}}
\DoxyCodeLine{57   \textcolor{comment}{/*                                                                       */}}
\DoxyCodeLine{58   \textcolor{comment}{/*       if ( FT\_ERR\_EQ( error, Invalid\_Outline ) )                      */}}
\DoxyCodeLine{59   \textcolor{comment}{/*         ...                                                           */}}
\DoxyCodeLine{60   \textcolor{comment}{/*                                                                       */}}
\DoxyCodeLine{61   \textcolor{comment}{/*     Using this macro you don't have to think about error prefixes.    */}}
\DoxyCodeLine{62   \textcolor{comment}{/*     Of course, if module errors are not active, the above example is  */}}
\DoxyCodeLine{63   \textcolor{comment}{/*     the same as                                                       */}}
\DoxyCodeLine{64   \textcolor{comment}{/*                                                                       */}}
\DoxyCodeLine{65   \textcolor{comment}{/*       if ( error == FT\_Err\_Invalid\_Outline )                          */}}
\DoxyCodeLine{66   \textcolor{comment}{/*         ...                                                           */}}
\DoxyCodeLine{67   \textcolor{comment}{/*                                                                       */}}
\DoxyCodeLine{68   \textcolor{comment}{/*   FT\_ERROR\_BASE( errcode )                                            */}}
\DoxyCodeLine{69   \textcolor{comment}{/*   FT\_ERROR\_MODULE( errcode )                                          */}}
\DoxyCodeLine{70   \textcolor{comment}{/*     Get base error and module error code, respectively.               */}}
\DoxyCodeLine{71   \textcolor{comment}{/*                                                                       */}}
\DoxyCodeLine{72   \textcolor{comment}{/*                                                                       */}}
